const mathOperation = (arg1, arg2, operation) => {
    if (
        (arg1 == 0 || arg1) &&
        (arg2 == 0 || arg2) &&
        operation &&
        !(operation === 'div' && arg2 === 0)
    ) {
        switch (operation) {
            case 'add':
                return +arg1 + +arg2;
            case 'sub':
                return arg1 - arg2;
            case 'mul':
                return arg1 * arg2;
            case 'div':
                return arg1 / arg2;
        }
    } else {
        return null;
    }
};

const power = (val, pow) => {
    if (pow <= 1) {
        return val;
    }
    return val * power(val, pow - 1);
};
